#include #include // Gebruik deze specifieke lange UUID's voor stabiliteit op iPad BLEService castService("19B10000-E8F2-537E-4F6C-D104768A1214"); BLECharacteristic dataChar("19B10001-E8F2-537E-4F6C-D104768A1214", BLENotify, 12); void setup() { IMU.begin(); BLE.begin(); BLE.setLocalName("Cast"); BLE.setAdvertisedService(castService); castService.addCharacteristic(dataChar); BLE.addService(castService); BLE.advertise(); } void loop() { BLEDevice central = BLE.central(); if (central && central.connected()) { unsigned long lastMillis = 0; while (central.connected()) { unsigned long currentMillis = millis(); if (currentMillis - lastMillis >= 20) { // 50Hz lastMillis = currentMillis; float ax, ay, az, gx, gy, gz; // Lees sensoren, stuur 0 als ze niet klaar zijn if (IMU.accelerationAvailable() && IMU.gyroscopeAvailable()) { IMU.readAcceleration(ax, ay, az); IMU.readGyroscope(gx, gy, gz); } else { ax=ay=az=gx=gy=gz=0; } // Pakket maken (12 bytes) int16_t packet[6]; packet[0] = (int16_t)(ax * 100); packet[1] = (int16_t)(ay * 100); packet[2] = (int16_t)(az * 100); packet[3] = (int16_t)(gx); packet[4] = (int16_t)(gy); packet[5] = (int16_t)(gz); dataChar.writeValue((byte*)packet, 12); } delay(1); } } }
